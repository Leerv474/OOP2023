15. Сравните чем отличается реализация массивов в С++ и С#?
Создадим массив из 10 элементов и заполним его случайными двухзначными числами на языке С++:
int Array[10];
srand(time(NULL)); //подключаем встряхивание случайных чисел
for (int i=0; i<10; i++){
Array[i] = rand()%90+10;
cout<<Array[i]<<" ";
}
cout<<endl;
Создадим массив из 10 элементов и заполним его случайными двухзначными числами на языке С# :
int[] Array = new int[10];
Random rnd = new Random(); //подключаем случайные числа
for (int i=0; i<10; i++)
{
Array[i] = rnd.Next(10,99);
Console.Write($"{Array[i]} ");
}
Console.WriteLine();
